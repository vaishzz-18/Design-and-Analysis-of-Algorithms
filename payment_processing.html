#include <iostream> #include <stack> #include <string> struct Payment { std::string vehicle; float cost; std::string mode_of_payment; }; // Example function to process payments void processPayments(std::stack<Payment>& payments) { while (!payments.empty()) { auto payment = payments.top(); std::cout << "Vehicle: " << payment.vehicle << ", Cost: " << payment.cost << ", Mode of Payment: " << payment.mode_of_payment << '\n'; payments.pop(); } }
