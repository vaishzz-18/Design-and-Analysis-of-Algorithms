#include <iostream> #include <queue> #include <string> struct Feedback { std::string username; float rating; std::string feedback; }; // Example function to display feedbacks void displayFeedbacks(const std::queue<Feedback>& feedbacks) { auto temp = feedbacks; while (!temp.empty()) { auto feedback = temp.front(); std::cout << "Username: " << feedback.username << ", Rating: " << feedback.rating << ", Feedback: " << feedback.feedback << '\n'; temp.pop(); } }
