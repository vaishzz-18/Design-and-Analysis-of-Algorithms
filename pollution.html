Pollution Node Example

#include <iostream> #include <memory> #include <string> struct PollutionNode { float p_level; std::string vehicle_type; std::shared_ptr<PollutionNode> next; }; // Example function to display pollution details void displayPollution(const std::shared_ptr<PollutionNode>& head) { auto current = head; while (current) { std::cout << "Vehicle: " << current->vehicle_type << ", Pollution Level: " << current->p_level << '\n'; current = current->next; } }
