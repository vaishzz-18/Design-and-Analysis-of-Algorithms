Sort Vehicles by Cost

#include <iostream> #include <vector> #include <algorithm> struct Vehicle { std::string type; float cost; }; // Example function to sort vehicles by cost void sortVehiclesByCost(std::vector<Vehicle>& vehicles) { std::sort(vehicles.begin(), vehicles.end(), [](const Vehicle& a, const Vehicle& b) { return a.cost < b.cost; }); for (const auto& vehicle : vehicles) { std::cout << "Vehicle: " << vehicle.type << ", Cost: " << vehicle.cost << '\n'; } }
